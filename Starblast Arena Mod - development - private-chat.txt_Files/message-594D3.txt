var logStats = ["wins","losses","misses","highStreak","bonus","streaksBroken"], locateShip = function(req, handler) {
  let args=req.replace(/^\s+/,"").replace(/\s+/," ").split(" "),id=Number(args[1]||"NaN");
  if (isNaN(id)) game.modding.terminal.error("Please specify a ship id to take action");
  else
  {
    let ship=game.findShip(id);
    if (!ship) game.modding.terminal.error("Requested ship not found!");
    else
    {
      try{typeof handler == "function" && handler(ship, id, args)}
      catch(e){game.modding.terminal.error("An error occured while taking action with the requested ship!")}
    }
  }
}, toStatString = function(i,t) {
  let stats = logStats.map(f => i[f] || 0);
  return Math.max(...stats)?((t+1)+". "+i.name+" ("+stats.join("/")+")"):"";
}, toRankString = function(stats,nLF) {
  nLF = nLF || ""
  let gt = (Array.isArray(stats)?stats:[]).map(toStatString).filter(i=> i).join(nLF);
  return (gt?nLF:"") + gt + (gt?nLF:"");
}, ranktext = function(nLF) {
  let lf = "\n"+(nLF?"\r\n\r":"");
  return "LEAVING SHIPS:" + toRankString(game.custom.leaves, lf) + lf + "RANKINGS:" + toRankString(leaderboard, lf) + lf + "DUEL HISTORY:" + lf + (Array.isArray(game.custom.history)?game.custom.history:[]).map((j,i) => `${i + 1}: ${j}`).join(lf);
}, infoName = function (ship) {
  return `${ship?.name || "ANONYMOUS"} (ID ${ship?.id || "Unknown"})`
}, restoreShip = function (ship, data) {
  Object.assign(ship.custom, data ?? newShipStats)
}, restoreRound = function (game, roundIndex) {
  let data = game.custom.matchData[roundIndex];
  if (!data) return false;
  for (let ship of game.ships) restoreShip(ship, data.list?.[ship.id]);
  game.custom.history = data.history;
  game.custom.leaves = data.leaves;
  return true
}

game.modding.commands.restore = function (req) {
  locateShip(req, function (ship, id, args) {
    let restoreID = +args[2];
    let data = game.findShip(restoreID);
    if (data) {
      data.custom.qualified = false;
      data.custom.ready = false;
      data.custom.restored = true;
      data.gameover({" ": " "});
      data = getShipStats(data);
    }
    else {
      data = game.custom.leaves.find(s => s.id === restoreID);
      if (!data) return game.modding.terminal.error("Restore ship not found!");
      game.custom.leaves.splice(game.custom.leaves.indexOf(data), 1)
    }
    ship.custom.qualified = true;
    restoreShip(ship, getShipStats({custom: data}, true));
    addWarn(ship, "Your results have been restored.");
    echo(infoName(ship) + " was restored with data from " + infoName(data));
  })
}

game.modding.commands.undo = function () {
  let test = restoreRound(game, game.custom.ArenaManager.rounds - 1);
  if (!test) return game.modding.terminal.error("No data to undo!");
  --game.custom.ArenaManager.rounds;
  echo("Undid the game to the previous round result");
}

game.modding.commands.redo = function () {
  let test = restoreRound(game, game.custom.ArenaManager.rounds + 1);
  if (!test) return game.modding.terminal.error("No data to redo!");
  ++game.custom.ArenaManager.rounds;
  echo("Redid the game to the previous round result");
}

game.modding.commands.exportRankings = function () {
  echo('BACKUP TEXT:');
  game.modding.commands.rankings();
  var element = document.createElement('a');
  element.setAttribute('href', 'data:text/plain;charset=utf-8,'+encodeURIComponent(ranktext(true)));
  element.setAttribute('download', "rankings_"+new Date().toLocaleDateString().replace(/\//g,"-"));

  element.style.display = 'none';
  document.body.appendChild(element);

  element.click();

  document.body.removeChild(element);
}
game.modding.commands.activate = function (bool) {
  game.custom.dev_started = bool.trim().split(" ")[1] == "true";
  if (game.custom.dev_started) {
    if (!game.custom.duration) game.custom.duration = 0;
    if (game.custom.dev_wait) game.custom.duration += game.step - game.custom.dev_wait;
    game.custom.dev_wait = game.step
  }
  else game.custom.paused_step = game.step;
  echo("The game is "+(game.custom.dev_started?"resumed":"paused"))
}
game.modding.commands.endgame = function (bool) {
  game.custom.ended = bool.trim().split(" ")[1] == "true";
  if (game.custom.ended) echo("The game will end after the current round has finished.");
  else echo("End game condition restored to default.")
}

    echo(ship.id+" : "+ship.name + " ("+logStats.map(i=>getStat(ship,i)).join("/")+"). "+(ship.custom.qualified?"":"Not ")+"Qualified.")

}

  locateShip(req, function (ship, id, args) {

    });
    echo(infoName(ship) + " was kicked");
  })
}

let NotifBox = class {
  constructor (ship, propName, color, position, max_messages, exposure_time) {
    if (!(ship.custom[propName] instanceof NotifBox)) {
      ship.custom[propName] = this;
      this.ship = ship;
      this.list = [];
      this.propName = propName;
      this.color = color;
      this.position = position;
      this.max_messages = max_messages;
      this.exposure_time = exposure_time
    }
    else Object.assign(this, ship.custom[propName])
  }

  add (message, permanent, allowDuplicates, align) {
    let oldLen = this.list.length;
    if (!allowDuplicates) {
      let ti;
      while (ti = this.list.findIndex(u => u?.[0] == message), ti != -1) this.list.splice(ti,1);
    }
    let rt = [message,permanent?-1:game.step,align];
    this.list.push(rt);
    this.update(oldLen)
  }

  remove (message, globalSearch) {
    let oldLen = this.list.length, ti;
    while (ti = this.list.findIndex(u => u[0] == message), ti != -1) {
      this.list.splice(ti,1);
      if (!globalSearch) break;
    }
    this.update(oldLen)
  }

  refresh () {
    return this.list = this.list.filter(Array.isArray).slice(-this.max_messages)
  }

  tick (game) {
    let oldLen = this.list.length;
    let i = 0;
    while (i < this.list.length) {
      let wt = this.list[i];
      if (typeof wt[1] != "number" || (wt[1] !== -1 && game.step - wt[1] > this.exposure_time*60)) this.list.splice(i,1);
      else ++i;
    }
    this.update(oldLen)
  }

  update (oldLen) {
    if (oldLen != this.list.length) this.show();
  }

  show () {
    this.refresh();
    let wh = 100 / this.max_messages;
    sendUI(this.ship, {
      id: this.propName,
      position: this.position,
      visible: true,
      clickable: false,
      components: this.list.map((message, i) => {
        let pos = wh * i;
        let content = message[0];
        if ("string" == typeof content) return {type: "text", position: [0, pos, 100, wh], value: content, color: this.color, align: message[2]}
        if (!Array.isArray(content)) return [];
        let contentLength = content.reduce((a,b) => a + b[0].length, 0), start = 0, res = [];
        for (let data of content) {
          let str = data[0];
          if ("string" == typeof str) {
            res.push({type: "text", position: [start / contentLength * 100, pos, str.length / contentLength * 100, wh], value: str, color: data[1] || this.color, align: data[2] || message[2]});
            start += str.length
          }
        }
        return res
      }).flat(Infinity)
    })
  }
}

    rad = Math.max(rad,0);
    percentage = Math.max(Math.min(percentage,1),0);
    if (!this.started || this.duration < 0) percentage = 0;
    return arena_radius - percentage * rad
  }

    });

      let shrinkSize = arena_radius - grad/10;

  }

    this.members.splice(0);
    this.duration = this.game_time;
    this.countdown = this.dc_time;
    this.shrink(game, init);
    this.angle = 2*Math.PI*Math.random();

  }

      let rando = rand(2);
      killer = this.members[rando];
      ship = this.members[1 - rando]
    }
    else if (!(this.members.indexOf(ship) != -1 && (killer = this.members.find(sus => sus != ship)))) return;

    addKillFeed([
      getPlayerRankArray(killer),
      [" beat ", null],
      getPlayerRankArray(ship)
    ]);

    setStates(killer, true, false, false);

    killer.custom.highStreak = Math.max(getStat(killer, 'streak'), getStat(killer, 'highStreak'));
    let shipStreak = getStat(ship, 'streak');
    if (shipStreak >= streakAnnouncement) {
      let bonusPoint = Math.trunc(shipStreak / 5);
      addWarn(killer, `You broke ${ship.name}'s win streak of 🔥${shipStreak}! +${bonusPoint + 1} to your streak breaker bonus!`);

        getPlayerRankArray(ship),
        [`'s win streak of 🔥${shipStreak}!`, streak_color]
      ])
    }
    let killerStreak = getStat(killer, 'streak');
    if (killerStreak >= streakAnnouncement) {
      let message;
      if (killerStreak % 5 === 0) {
        let words = streakMessages[killerStreak / 5];

      else message = `has a win streak of 🔥${killerStreak}`;
      addKillFeed([
        getPlayerRankArray(killer),
        [` ${message}!`, streak_color]
      ]);
    }

    ship.custom.highStreak = Math.max(getStat(ship, 'streak'), getStat(ship, 'highStreak'));
    ship.custom.streak = 0;
    ship.custom.losses = getStat(ship, 'losses') + 1;
    setStates(ship, false, true, false);
    ship.custom.isLastLost = true;

    updatescoreboard(game);

    this.arenaShrinkMessage(false);

    let names = `${infoName(killer)} beat ${infoName(ship)}`;

    if (!game.custom.history) game.custom.history = [names];
    else game.custom.history.push(names);

    this.members.forEach(aship => resetShip(aship, true));

    this.countdown = this.duration = 0;

    modUtils.setTimeout(this.reset.bind(this), pendingTpDelay * 60)
  }


  tick (game) {
    if (this.inCountdown) {
      this.countdown--;
      if (this.inMatch) this.members.forEach(ship => {
        max(ship, ship.custom.type);
        ship.set({idle: false, generator: 1e5, invulnerable: 0, crystals: (gem_ratio === false)?fixed_gems:20*(Math.trunc(ship.custom.type/100)**2)*gem_ratio});
      })
    }
    else if (this.inMatch) {
      let duration = this.duration--;
      if ((min_radius_ratio === false ? fixed_min_radius < arena_radius : min_radius_ratio < 1) && this.started && !this.ended && duration <= toTick(shrink_start_time) && duration >= toTick(shrink_end_time)) {
        if (game.step % (shrink_interval*60) === 0) this.shrink(game);
        else if (duration == toTick(shrink_end_time)) this.shrink(game);
        this.arenaShrinkMessage(true)
      }
      else this.arenaShrinkMessage(false);
      if (this.ended) this.endDuel(true);
    }
  }

  get inCountdown () {
    return this.started && this.countdown > 0
  }

  get inMatch () {
    return this.started && this.countdown <= 0 && this.duration > 0
  }

  get isAvailable () {
    return !(this.lobby || this.started || this.ended)
  }

  get ended () {
    return this.started && !this.inCountdown && !this.inMatch
  }

  dc_time = toTick(duel_countdown/60);

  game_time = toTick(duel_duration);
}

  checkPlayers: function(game) {
    let waiting_list = game.ships.filter(ship => ship.custom.isReady() && !ship.custom.paired);
    if (this.isOpen(game)) {
      let arenaSlot;
      if (game.custom.dev_started && this.isAvailable() && waiting_list.length > 1) {
        if (this.waiting_time < 0) {
          this.waiting_time = this.queueWaitingTime * 60 + 45;
          if (this.game_rounds != false) {
            let roundsLeft = this.game_rounds - this.rounds;
            if ([10, 5, 1].indexOf(roundsLeft) != -1) game.ships.forEach(s => addWarn(s, `${roundsLeft} round${roundsLeft != 1 ? "s" : ""} left!`))
          }
        }
        else --this.waiting_time;
        if (this.waiting_time < 0) {
          waiting_list.sort((a,b) => {
            let dwins = (getStat(b, 'wins') + getStat(b, 'bonus')) - (getStat(a, 'wins') + getStat(a, 'bonus'));
            if (dwins !== 0) return dwins;
            let dlosses = getStat(a, 'losses') - getStat(b, 'losses');
            if (dlosses !== 0) return dlosses;
            let ddraws = getStat(b, 'draws') - getStat(a, 'draws');
            if (ddraws !== 0) return ddraws;
            let dmisses = getStat(b, 'misses') - getStat(a, 'misses');
            if (dmisses !== 0) return dmisses;
            let lastStatusRank = (a.custom.isLastLost != b.custom.isLastLost) ? (a.custom.isLastLost - b.custom.isLastLost) : ((a.custom.isLastLost ? -1 : 1) * (b.custom.lastMatchStep - a.custom.lastMatchStep));
            return isNaN(lastStatusRank) ? (a.id - b.id) : lastStatusRank
          });
          let highRanksPlayersCount = Math.trunc(waiting_list.length * highRanksPercentage / 100 / 2) * 2;
          let highRanks = waiting_list.slice(0, highRanksPlayersCount), lowRanks = waiting_list.slice(highRanksPlayersCount);
          while ((arenaSlot = this.findAvailable()) && waiting_list.length > 1) {
            let list = getPair(highRanks.length > 0 ? highRanks : lowRanks);
            for (let i = 0; i < 2; ++i) list[i].custom.lastOpponentID = list[1 - i].id;
            arenaSlot.assign(...list);
            waiting_list = [...highRanks, ...lowRanks];
          }
          waiting_list.forEach(s => {
            s.custom.misses = getStat(s, 'misses') + 1;
            s.custom.missed = true;
            addWarn(s, "No opponents found!")
          });
          ++this.rounds;
        }
      }
      else this.waiting_time = -1;
      // while ((arenaSlot = this.findAvailable()) && this.invites.length > 0) arenaSlot.assign(...this.invites.splice(0, 1)[0])
    }
    else this.waiting_time = -1;
    this.waiting_count = waiting_list.length// + this.invites.length * 2;
    this.list.forEach(arena => !arena.isAvailable && arena.tick(game))
  },
  isOpen: function (game) { return this.isAvailable() && !game.custom.ended && (!game.custom.dev_started || (this.game_duration === false ? (this.rounds < this.game_rounds) : ((game.custom.duration || 0) + game.step - (game.custom.dev_wait || 0)) <= toTick(this.game_duration + duel_duration / 2)))},
  isRunning: function (game) {return this.isOpen(game) || !this.isAvailable() },
  set: function(ship, node, forced, reversed) {
    let nodpos = this.list[node] || this.lobby;
    if (ship != null) {
      if (!ship.custom.spectate) {
        ship.custom.arena = nodpos;
        ship.custom.TpTimestamp = game.step;
      }
      if (ship.custom.pendingTp === -1 || forced) {
        ship.custom.pendingTp = -1;
        let angle = 0, distance = 0, noded = !nodpos.lobby && !ship.custom.spectate;
        if (noded) {
          angle = nodpos.angle || 0;
          if (reversed) angle += Math.PI;
          distance = 2/3*r;
        }
        let {x,y} = nodpos, setup = {x:x+distance*Math.cos(angle),y:y+distance*Math.sin(angle)};
        if (noded) setup.angle = 180*((1 + angle/Math.PI)%2);
        ship.set(setup);
      }
    }
  },
  findNearest: function (ship) {
    let t = this.list.map(i => i.distTo(ship)), min = Math.min(...t);
    return this.list[t.indexOf(min)]
  }
